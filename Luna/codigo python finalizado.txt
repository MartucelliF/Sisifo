import mysql.connector
import serial
import time

connection = mysql.connector.connect(
    host="localhost",
    user="root",
    passwd="",
    database="sensores_db"
)
cur = connection.cursor()

cur.execute("SELECT * FROM sensores LIMIT 5")
respuesta = cur.fetchall()

# Formatear los datos como una cadena
datos_formateados = "\n".join([str(fila) for fila in respuesta])

arduino = serial.Serial('COM6', 9600)
time.sleep(3)


arduino.write(datos_formateados.encode())
print("Datos enviados al Arduino.")
